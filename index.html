<script>

    function calculatePosition() {

        const totalCap = parseFloat(document.getElementById('totalCap').value);

        const riskPct = parseFloat(document.getElementById('riskPct').value) / 100;

        const entryPrice = parseFloat(document.getElementById('entryPrice').value);

        const stopPrice = parseFloat(document.getElementById('stopPrice').value);


        if (totalCap && riskPct && entryPrice && stopPrice && entryPrice !== stopPrice) {

            const riskAmt = totalCap * riskPct;

            const priceDiff = Math.abs(entryPrice - stopPrice);

            const totalQty = (riskAmt / priceDiff).toFixed(2);


            document.getElementById('calcResult').innerHTML = `

                最大允许损失额度: <span class="result-highlight">${riskAmt.toLocaleString()}</span><br>

                <strong>建议总买入数量: <span class="result-highlight">${totalQty}</span></strong> (由风险数学决定)

            `;


            document.getElementById('s1').innerText = (totalQty * 0.2).toFixed(2);

            document.getElementById('s2').innerText = (totalQty * 0.6).toFixed(2);

            document.getElementById('s3').innerText = (totalQty * 0.2).toFixed(2);

        }

    }
